#include using namespace std; class Node { public: int data; Node* next; Node(int val) { data = val; next = NULL; } }; void insertAtBeginning(Node* &head, int val) { Node* newNode = new Node(val); newNode->next = head; head = newNode; } void insertAtEnd(Node* &head, int val) { Node* newNode = new Node(val); if (head == NULL) { head = newNode; return; } Node* temp = head; while (temp->next != NULL) { temp = temp->next; } temp->next = newNode; } void insertAfterValue(Node* &head, int key, int val) { Node* temp = head; while (temp != NULL && temp->data != key) { temp = temp->next; } if (temp == NULL) { cout << "Value not found!" ; return; } Node* newNode=new Node(val); newNode->next = temp->next; temp->next = newNode; } void display(Node* head) { Node* temp = head; while (temp != NULL) { cout << temp->data << " -> " ; temp=temp->next; } cout << "NULL" ; } int main() { Node* head=NULL; insertAtBeginning(head, 10); insertAtBeginning(head, 5); insertAtEnd(head, 20); insertAtEnd(head, 30); insertAfterValue(head, 20, 25); display(head); return 0; }
